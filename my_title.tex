		colframe=black!30!white,

			\begin{tcbclipinterior}
			\fill[black!30!white]
				(frame.south west)
					rectangle node[text=white,font=\sffamily\bfseries\footnotesize,rotate=90] {FILE} 
				([xshift=4mm]frame.north west);

		}
	},
